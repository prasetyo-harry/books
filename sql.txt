SELECT C.NAME
FROM CUSTOMER C
LEFT JOIN ORDER_TABLE O ON C.ID = O.CUSTOMER_ID
WHERE O.ID IS NULL;

/**  OPTION 2 **/
SELECT C.NAME
FROM CUSTOMER C
WHERE C.ID NOT IN (SELECT DISTINCT CUSTOMER_ID FROM ORDER_TABLE);

SELECT nim, kode_matakuliah, COUNT(*) AS jumlah_duplikasi
FROM mahasiswa
GROUP BY nim, kode_matakuliah
HAVING COUNT(*) > 1;
